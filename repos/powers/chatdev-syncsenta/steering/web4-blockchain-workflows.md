   // Generated by ChatDev
   contract SyncSentaCredentials is ERC721, AccessControl, Pausable {
       bytes32 public constant ISSUER_ROLE = keccak256("ISSUER_ROLE");
       
       struct Credential {
           address learner;
           string skillId;
           string evidenceHash;  // IPFS CID
           uint256 issuedAt;
           bool revoked;
       }
       
       mapping(uint256 => Credential) public credentials;
       uint256 private _tokenIdCounter;
       
       function mintCredential(
           address learner,
           string memory skillId,
           string memory evidenceHash
       ) external onlyRole(ISSUER_ROLE) whenNotPaused returns (uint256) {
           uint256 tokenId = _tokenIdCounter++;
           _safeMint(learner, tokenId);
           credentials[tokenId] = Credential({
               learner: learner,
               skillId: skillId,
               evidenceHash: evidenceHash,
               issuedAt: block.timestamp,
               revoked: false
           });
           emit CredentialMinted(tokenId, learner, skillId);
           return tokenId;
       }
       
       function verifyCredential(uint256 tokenId)
           external view returns (bool valid, address learner, string memory skillId) {
           require(_exists(tokenId), "Credential does not exist");
           Credential memory cred = credentials[tokenId];
           return (!cred.revoked, cred.learner, cred.skillId);
       }
       
       function revokeCredential(uint256 tokenId) external onlyRole(ISSUER_ROLE) {
           require(_exists(tokenId), "Credential does not exist");
           credentials[tokenId].revoked = true;
           emit CredentialRevoked(tokenId);
       }
   }
   ```

   // Generated by ChatDev
   use ipfs_api::{IpfsClient, IpfsApi};
   use cid::Cid;
   
   pub struct IPFSService {
       client: IpfsClient,
       pinata_api_key: String,
       pinata_secret_key: String,
   }
   
   impl IPFSService {
       pub async fn upload_content(&self, data: Vec<u8>) -> Result<Cid, IPFSError> {
           // Upload to local IPFS node
           let response = self.client.add(data.clone()).await?;
           let cid = Cid::try_from(response.hash.as_str())?;
           
           // Pin to Pinata for persistence
           self.pin_to_pinata(&cid).await?;
           
           Ok(cid)
       }
       
       pub async fn retrieve_content(&self, cid: &Cid) -> Result<Vec<u8>, IPFSError> {
           // Try local node first
           match self.client.cat(&cid.to_string()).await {
               Ok(data) => Ok(data),
               Err(_) => {
                   // Fallback to public gateway
                   self.fetch_from_gateway(cid).await
               }
           }
       }
   }
   ```

2. **Pinning Service Integration**
   - Pinata for primary pinning
   - Infura as backup
   - Local node for development

3. **Gateway Fallback**
   - Try local node first
   - Fall back to public gateways
   - Multiple gateway options for redundancy

4. **Content Addressing**
   - All content referenced by CID
   - Immutable content
   - Verifiable integrity

   // Generated by ChatDev
   contract SyncToken is ERC20, AccessControl {
       bytes32 public constant MINTER_ROLE = keccak256("MINTER_ROLE");
       
       mapping(address => uint256) public earnedTokens;
       mapping(address => uint256) public redeemedTokens;
       
       function mint(address to, uint256 amount) external onlyRole(MINTER_ROLE) {
           _mint(to, amount);
           earnedTokens[to] += amount;
           emit TokensMinted(to, amount);
       }
       
       function burn(uint256 amount) external {
           _burn(msg.sender, amount);
           redeemedTokens[msg.sender] += amount;
           emit TokensBurned(msg.sender, amount);
       }
       
       function getEarningHistory(address learner)
           external view returns (uint256 earned, uint256 redeemed, uint256 balance) {
           return (earnedTokens[learner], redeemedTokens[learner], balanceOf(learner));
       }
   }

   ```solidity
   // Generated by ChatDev
   contract SyncSentaDAO {
       struct Proposal {
           uint256 id;
           address proposer;
           string description;
           string curriculumChange;
           uint256 votesFor;
           uint256 votesAgainst;
           uint256 deadline;
           bool executed;
           mapping(address => bool) hasVoted;
       }
       
       mapping(uint256 => Proposal) public proposals;
       uint256 public proposalCount;
       
       function propose(string memory description, string memory curriculumChange)
           external returns (uint256) {
           require(syncToken.balanceOf(msg.sender) >= MIN_PROPOSAL_TOKENS, "Insufficient tokens");
           
           uint256 proposalId = proposalCount++;
           Proposal storage proposal = proposals[proposalId];
           proposal.id = proposalId;
           proposal.proposer = msg.sender;
           proposal.description = description;
           proposal.curriculumChange = curriculumChange;
           proposal.deadline = block.timestamp + VOTING_PERIOD;
           
           emit ProposalCreated(proposalId, msg.sender, description);
           return proposalId;
       }
       
       function vote(uint256 proposalId, bool support) external {
           Proposal storage proposal = proposals[proposalId];
           require(block.timestamp < proposal.deadline, "Voting ended");
           require(!proposal.hasVoted[msg.sender], "Already voted");
           
           uint256 votes = syncToken.balanceOf(msg.sender);
           
           if (support) {
               proposal.votesFor += votes;
           } else {
               proposal.votesAgainst += votes;
           }
           
           proposal.hasVoted[msg.sender] = true;
           emit Voted(proposalId, msg.sender, support, votes);
       }
       
       function execute(uint256 proposalId) external {
           Proposal storage proposal = proposals[proposalId];
           require(block.timestamp >= proposal.deadline, "Voting not ended");
           require(!proposal.executed, "Already executed");
           require(proposal.votesFor > proposal.votesAgainst, "Proposal rejected");
           
           // Execute curriculum change
           curriculumRegistry.updateCurriculum(proposal.curriculumChange);
           proposal.executed = true;
           
           emit ProposalExecuted(proposalId);
       }
   }
   ```

## Best Practices for Web4 Development

### 1. Decentralization-First Design

Always prioritize decentralization:

```bash
# Wrong: Centralized storage
python ChatDev/run.py --task "Store content in S3"

# Right: Decentralized storage
python ChatDev/run.py --task "Store content on IPFS with Pinata pinning"
```

### 2. User Sovereignty

Users own their identity and data:

```bash
python ChatDev/run.py \
  --task "Implement feature" \
  --principle "user-sovereignty" \
  --did-based-auth \
  --user-controlled-data
```

### 3. Censorship Resistance

Design for censorship resistance:

```bash
python ChatDev/run.py \
  --task "Implement content distribution" \
  --storage "ipfs" \
  --no-single-point-of-failure \
  --multiple-gateways
```

### 4. Blockchain for Trust

Use blockchain for trust-critical operations:

```bash
python ChatDev/run.py \
  --task "Implement credential system" \
  --on-chain "credentials,approvals,governance" \
  --off-chain "content,metadata"
```

### 5. Gas Optimization

Optimize for gas efficiency:

```bash
python ChatDev/run.py \
  --task "Optimize smart contract" \
  --agents programmer_blockchain,reviewer \
  --optimize-gas \
  --target-gas-cost "< 100000"
```
